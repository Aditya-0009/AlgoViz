N-Queen Problem

// N-Queen Algorithm Implementation in C++

using namespace std;

bool isSafe(vector>& board, int row, int col, int n) {
    for (int i = 0; i < col; i++)
        if (board[row][i])
            return false;

    for (int i = row, j = col; i >= 0 && j >= 0; i--, j--)
        if (board[i][j])
            return false;

    for (int i = row, j = col; j >= 0 && i < n; i++, j--)
        if (board[i][j])
            return false;

    return true;
}

bool solveNQUtil(vector>& board, int col, int n) {
    if (col >= n)
        return true;

    for (int i = 0; i < n; i++) {
        if (isSafe(board, i, col, n)) {
            board[i][col] = 1;
            if (solveNQUtil(board, col + 1, n))
                return true;
            board[i][col] = 0;
        }
    }

    return false;
}

void solveNQ(int n) {
    vector> board(n, vector(n, 0));
    if (!solveNQUtil(board, 0, n)) {
        cout << "Solution does not exist";
        return;
    }
    // Print the solution
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            cout << board[i][j] << " ";
        cout << endl;
    }
}

int main() {
    int n = 4; // Example size
    solveNQ(n);
    return 0;
}

Description

The N-Queen problem is a classic combinatorial problem that involves placing N queens on an N x N chessboard such that no two queens threaten each other. This means that no two queens can be placed in the same row, column, or diagonal.

The algorithm visualized here attempts to place queens on the board one column at a time, checking for safety before placing a queen. It uses a backtracking approach to explore all possible solutions, ensuring that each solution satisfies the constraints of the problem.

Complexity

Time Complexity:

  • Best Case: O(n2) when the solution is found early.
  • Average Case: O(n2).
  • Worst Case: O(n2).

Space Complexity: O(n2) due to storing the board configuration.
